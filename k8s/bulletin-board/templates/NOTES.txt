1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
  export INGRESS_IP=$(kubectl get ingress {{ include "bulletin-board.fullname" . }} \
    -n {{ include "bulletin-board.namespace" . }} \
    -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "Visit http://$INGRESS_IP{{ .Values.ingress.path }}"
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ include "bulletin-board.namespace" . }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "bulletin-board.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ include "bulletin-board.namespace" . }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  kubectl get --namespace {{ include "bulletin-board.namespace" . }} svc -w {{ include "bulletin-board.fullname" . }}
{{- else if contains "ClusterIP" .Values.service.type }}
  kubectl port-forward --namespace {{ include "bulletin-board.namespace" . }} svc/{{ include "bulletin-board.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}
  echo "Visit http://127.0.0.1:{{ .Values.service.port }}"
{{- end }}

2. Yandex Cloud auth secret for Lockbox:
   make secret_file    # k8s/authorized-key.json (auto on helm-install)
   make secrets-values # k8s/bulletin-board/values.secrets.yaml
   make helm-install

3. Update externalSecrets.lockboxSecretId in values.yaml after terraform apply:
   terraform -chdir=terraform output -raw lockbox_secret_id

4. External access via Ingress (if enabled):
   make install-ingress
   kubectl get ingress -n {{ include "bulletin-board.namespace" . }}
